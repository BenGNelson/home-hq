# Source of intent for the backend's Python deps: human-readable, commented,
# loose floors. The image actually installs the fully-pinned resolution in
# requirements.lock (reproducible rebuilds). After bumping a floor here,
# regenerate the lock: docker compose run --rm --no-deps backend pip freeze > requirements.lock

# Web framework + ASGI server that runs it
fastapi>=0.110
uvicorn[standard]>=0.29

# Reads config from env / .env into a typed settings object
pydantic-settings>=2.2

# System stats (cpu/ram/disk/uptime) for /api/system and /api/disk
psutil>=5.9

# Docker SDK for /api/containers (added when we build that endpoint)
docker>=7.0

# Plex client for /api/plex.
# NOTE: PyPI package is "PlexAPI"; the Python import is "plexapi".
PlexAPI>=4.15

# MQTT client for /api/printer — the Bambu P1S publishes telemetry over a local
# MQTT broker in LAN mode. We keep a persistent background connection and cache
# the latest snapshot (the only push-based source; everything else is pull).
paho-mqtt>=2.1

# Multipart form parsing — used by the Library save-state upload (the emulator
# POSTs a state blob + screenshot as multipart/form-data).
python-multipart>=0.0.18  # >=0.0.18 fixes CVE-2024-53981 (multipart DoS)

# Image processing — downscales + re-encodes box art / Plex posters to small
# WebP thumbnails so the art grids load fast (see app/images.py).
Pillow>=10.3

# Comic archive reading — CBZ/CBR/CB7 are zip/rar/7z archives of page images.
# libarchive handles all of them through one binding (needs the system
# libarchive13 lib, installed in the Dockerfile). See app/comics.py.
libarchive-c>=5.1

# Audio metadata — pulls an audiobook's embedded cover art (ID3 APIC / MP4 covr /
# FLAC pictures) out of its first chapter. Pure Python, no system libs.
mutagen>=1.47

# PDF rendering — magazines/papers and PDF books carry no embedded cover, so we
# render their first page as the cover (app/pdfcover.py). Ships a self-contained
# MuPDF build in the wheel, so no system libs are needed.
PyMuPDF>=1.27

# Enphase Envoy (solar) local API client for /api/solar. Handles the firmware 7+
# token auth (mint + ~6-month auto-refresh via the Enlighten cloud), the
# self-signed local HTTPS, and the metered/non-metered data model. Same library
# Home Assistant's official Envoy integration is built on. See app/solar.py.
pyenphase>=1.20

# HTTP client used directly by /api/weather (Open-Meteo) and the Plex art proxy.
# It already arrives transitively via PlexAPI/docker, but weather.py depends on it
# directly so we pin it explicitly.
requests>=2.31
